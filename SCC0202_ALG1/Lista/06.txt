	empilha(C, x);
	x = desempilha(A);
	empilha(B, x);
	x = desempilha(C);
	empilha(B, x);
	x = desempilha(A);
	empilha(C, x);
	x = desempilha(B);
	empilha(A, x);
	x = desempilha(B);
	empilha(C, x);
	x = desempilha(A);
	empilha(C, x);
